    <script>
        // --- BASE DE DATOS (Ejemplo con 3, aquí irían los 72) ---
        const namesOfGod = [
            { 
                hebrew: "והו", 
                trans: "Vehuiah", 
                meaning: "Voluntad y creación nueva. Ayuda a emprender cosas nuevas y difíciles. Otorga una voluntad poderosa." 
            },
            { 
                hebrew: "ילי", 
                trans: "Jeliel", 
                meaning: "Amor y sabiduría. Calma las revueltas populares y restablece la paz en las parejas. Simboliza la fecundidad." 
            },
            { 
                hebrew: "סיט", 
                trans: "Sitael", 
                meaning: "Construcción del universo. Protege contra la adversidad y ayuda a descubrir la verdad. Constructor de mundos." 
            },
            { 
                hebrew: "עלם", 
                trans: "Elemiah", 
                meaning: "Poder Divino. Ayuda a evitar crisis en empresas y protege en los viajes. Otorga paz interior." 
            },
            { 
                hebrew: "מהש", 
                trans: "Mahasiah", 
                meaning: "Curación y rectificación. Capacidad para vivir en paz con todo el mundo. Rectifica el carácter." 
            }
        ];

        // --- LÓGICA DEL SISTEMA ---

        const cardFan = document.getElementById('card-fan');
        const activeStage = document.getElementById('active-stage');
        const resetBtn = document.getElementById('reset-btn');
        
        // Elementos de la carta activa
        const uiHebrew = document.getElementById('card-hebrew');
        const uiTrans = document.getElementById('card-trans');
        const uiMeaning = document.getElementById('card-meaning');
        const cardInner = document.querySelector('.card-inner');

        // Función para barajar (Algoritmo Fisher-Yates)
        function shuffle(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
            return array;
        }

        // Renderizar el abanico
        function renderFan() {
            cardFan.innerHTML = '';
            // Mezclamos el array para que siempre sea distinto
            const deck = shuffle([...namesOfGod]); 
            
            // Cantidad de cartas a mostrar en el abanico (para demo visual)
            // En la app real, podrías mostrar 20-30 para no saturar o agrupar por Sefirot.
            const totalCards = deck.length; 
            const angleStep = 10; // Grados entre cartas
            const baseAngle = -((totalCards - 1) * angleStep) / 2;

            deck.forEach((data, index) => {
                const el = document.createElement('div');
                el.className = 'mini-card';
                
                // Cálculo de rotación para el abanico
                const rotation = baseAngle + (index * angleStep);
                
                el.style.transform = `rotate(${rotation}deg) translateY(${Math.abs(rotation)*0.5}px)`;
                
                // Evento al pulsar
                el.onclick = () => selectCard(data);
                
                cardFan.appendChild(el);
            });
        }

        // Seleccionar una carta
        function selectCard(data) {
            // Ocultar abanico
            cardFan.style.display = 'none';
            document.querySelector('h3').style.display = 'none';
            
            // Rellenar datos
            uiHebrew.innerText = data.hebrew;
            uiTrans.innerText = data.trans;
            uiMeaning.innerText = data.meaning;
            
            // Mostrar escenario
            activeStage.style.display = 'block';
            resetBtn.style.display = 'block';

            // Asegurar que empieza cara frontal
            cardInner.classList.remove('flipped');
        }

        // Dar la vuelta a la carta
        function flipCard() {
            cardInner.classList.toggle('flipped');
        }

        // Reiniciar
        function resetApp() {
            activeStage.style.display = 'none';
            resetBtn.style.display = 'none';
            cardFan.style.display = 'block';
            document.querySelector('h3').style.display = 'block';
            
            // Volver a generar el abanico (cambia el orden)
            renderFan();
        }

        // Iniciar al cargar
        renderFan();

    </script>
